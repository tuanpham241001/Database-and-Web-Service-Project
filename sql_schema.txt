CREATE TABLE Students(
    sid CHAR(8)
    name CHAR(30),
    mat_num CHAR(8) UNIQUE,
    birthday CHAR(8),
    rsid CHAR (8),              
    rid CHAR (8) NOT NULL,      
    PRIMARY KEY sid,
    FOREIGN KEY (rsid) REFERENCE Students (sid),
    FOREIGN KEY (rid) REFERENCE Rooms
)

//ISA Hierachy

CREATE TABLE Students_with_special_need(
    sickness CHAR(12),
    special_need CHAR(30),
    sid CHAR(8),
    PRIMARY KEY sid,
    FOREIGN KEY sid REFERENCE Student
)


CREATE TABLE Colleges(
    cid CHAR(8),
    name CHAR(12),
    adress CHAR(30),
    PRIMARY KEY cid
)

CREATE TABLE Rooms(
    rid CHAR(8),
    number CHAR(5)
    floor INT,
    mailbox CHAR(5),
    availabilty BIT,
    cid CHAR(8) NOT NULL,       
    PRIMARY KEY rid, 
    FOREIGN KEY cid REFERENCE College, 
)

//ISA Hierachy

CREATE TABLE Double_rooms(
    rid CHAR(8),
    position BIT,              
    PRIMARY KEY rid,
    FOREIGN KEY rid REFERENCE Rooms
)


CREATE TABLE Managers(
    mid CHAR(8),
    name CHAR(30),
    age INT,
    contact_num CHAR(12), 
    cid CHAR(8),                
    PRIMARY KEY mid,
    FOREIGN KEY (cid) REFERENCE College,
)

//ISA Hierachy

CREATE TABLE RA(
    availabilty BIT,
    mid CHAR(8),
    PRIMARY KEY mid,
    FOREIGN KEY mid REFERENCE Managers   
)

CREATE TABLE RM(
    office_hour CHAR(30),
    mid CHAR(8),
    PRIMARY KEY mid,
    FOREIGN KEY mid REFERENCE Managers
)
